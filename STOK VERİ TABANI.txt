UYELER
  
CREATE TABLE "HK"."UYELER" 
   (	"POZISYON" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"KADI" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"PAROLA" VARCHAR2(12 BYTE) NOT NULL ENABLE, 
	"MAIL" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	 PRIMARY KEY ("MAIL"));

URUNLER

CREATE TABLE "HK"."URUNLER" 
   (	"KOD" NUMBER(11,0) NOT NULL ENABLE, 
	"KATEGORI" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"URUNADI" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"BEDEN" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"RENK" VARCHAR2(30 BYTE) NOT NULL ENABLE, 
	"FIYAT" NUMBER NOT NULL ENABLE, 
	"ADET" NUMBER(4,0) NOT NULL ENABLE, 
	"ACIKLAMA" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	 PRIMARY KEY ("KOD"));

SATISLAR

CREATE TABLE "HK"."SATISLAR" 
   (	"KOD" NUMBER(11,0) NOT NULL ENABLE, 
	"KATEGORI" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"URUNADI" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"BEDEN" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"RENK" VARCHAR2(30 BYTE) NOT NULL ENABLE, 
	"FIYAT" NUMBER NOT NULL ENABLE, 
	"ADET" NUMBER(4,0) NOT NULL ENABLE, 
	"TOPLAM_FIYAT" NUMBER GENERATED ALWAYS AS ("FIYAT"*"ADET") VIRTUAL VISIBLE , 
	"ACIKLAMA" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	 PRIMARY KEY ("KOD"));
